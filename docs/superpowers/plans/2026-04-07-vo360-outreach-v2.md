# VO360 Outreach V2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Next.js web app that imports BatchLeads CSV exports, AI-scores leads for remodeling potential, generates personalized SMS messages, and sends them via GoHighLevel.

**Architecture:** Next.js 15 App Router with 4 pages (Upload, Review, Send, Log). CSV data flows through React state (no persistence until send). SQLite stores send log only. Server-side API routes handle Claude AI calls and GHL API calls.

**Tech Stack:** Next.js 15, Tailwind CSS, better-sqlite3, @anthropic-ai/sdk, GoHighLevel REST API

```ts
  "last sale date": "lastSaleDate",
  "last_sale_date": "lastSaleDate",
  "last sale price": "lastSalePrice",
  "last_sale_price": "lastSalePrice",
  "absentee owner": "absenteeOwner",
  "absentee_owner": "absenteeOwner",
  "free and clear": "freeAndClear",
  "free_and_clear": "freeAndClear",
};

function normalizeColumnName(col: string): string {
  return col.trim().toLowerCase();
}

function parseNumber(val: string | undefined): number | null {
  if (!val || val.trim() === "") return null;
  const cleaned = val.replace(/[$,%]/g, "").trim();
  const num = Number(cleaned);
  return isNaN(num) ? null : num;
}

function parseBoolean(val: string | undefined): boolean | null {
  if (!val || val.trim() === "") return null;
  const lower = val.trim().toLowerCase();
  if (["yes", "true", "1", "y"].includes(lower)) return true;
  if (["no", "false", "0", "n"].includes(lower)) return false;
  return null;
}

export function parseLeadsCSV(csvText: string): Lead[] {
  if (!csvText.trim()) return [];

  const parsed = Papa.parse(csvText.trim(), {
    header: true,
    skipEmptyLines: true,
  });

  if (!parsed.data || parsed.data.length === 0) return [];

  const headers = parsed.meta.fields || [];
  const columnMapping: Record<string, keyof Lead> = {};

  for (const header of headers) {
    const normalized = normalizeColumnName(header);
    if (COLUMN_MAP[normalized]) {
      columnMapping[header] = COLUMN_MAP[normalized];
    }
  }

  return (parsed.data as Record<string, string>[]).map((row) => {
    const mapped: Record<string, string> = {};
    for (const [csvCol, leadField] of Object.entries(columnMapping)) {
      mapped[leadField] = row[csvCol] || "";
    }

    return {
      firstName: mapped.firstName || "",
      lastName: mapped.lastName || "",
      phone: mapped.phone || "",
      propertyAddress: mapped.propertyAddress || "",
      city: mapped.city || "",
      state: mapped.state || "",
      zip: mapped.zip || "",
      propertyType: mapped.propertyType || "",
      bedrooms: parseNumber(mapped.bedrooms),
      bathrooms: parseNumber(mapped.bathrooms),
      sqft: parseNumber(mapped.sqft),
      yearBuilt: parseNumber(mapped.yearBuilt),
      estimatedValue: parseNumber(mapped.estimatedValue),
      equityPercent: parseNumber(mapped.equityPercent),
      ownerOccupied: parseBoolean(mapped.ownerOccupied),
      lastSaleDate: mapped.lastSaleDate || "",
      lastSalePrice: parseNumber(mapped.lastSalePrice),
      absenteeOwner: parseBoolean(mapped.absenteeOwner),
      freeAndClear: parseBoolean(mapped.freeAndClear),
    };
  });
}
```

```ts
const GHL_BASE = "https://services.leadconnectorhq.com";
const GHL_VERSION = "2021-07-28";

function getHeaders(): Record<string, string> {
  const apiKey = process.env.GHL_API_KEY;
  if (!apiKey) throw new Error("GHL_API_KEY not set");
  return {
    Authorization: `Bearer ${apiKey}`,
    Version: GHL_VERSION,
    "Content-Type": "application/json",
  };
}

function getLocationId(): string {
  const locationId = process.env.GHL_LOCATION_ID;
  if (!locationId) throw new Error("GHL_LOCATION_ID not set");
  return locationId;
}

export async function searchContactByPhone(
  phone: string
): Promise<{ id: string; firstName: string } | null> {
  const res = await fetch(
    `${GHL_BASE}/contacts/search/duplicate?locationId=${getLocationId()}&number=${encodeURIComponent(phone)}`,
    { headers: getHeaders() }
  );

  if (!res.ok) {
    const text = await res.text();
    throw new Error(`GHL search failed (${res.status}): ${text}`);
  }

  const data = await res.json();
  const contact = data.contact;
  if (contact && contact.id) {
    return { id: contact.id, firstName: contact.firstName || "" };
  }
  return null;
}

export async function createContact(params: {
  firstName: string;
  lastName: string;
  phone: string;
  address1?: string;
  city?: string;
  state?: string;
  postalCode?: string;
}): Promise<string> {
  const res = await fetch(`${GHL_BASE}/contacts/`, {
    method: "POST",
    headers: getHeaders(),
    body: JSON.stringify({
      locationId: getLocationId(),
      firstName: params.firstName,
      lastName: params.lastName,
      phone: params.phone,
      address1: params.address1 || "",
      city: params.city || "",
      state: params.state || "",
      postalCode: params.postalCode || "",
      tags: ["vo360-outreach"],
      source: "vo360-outreach-v2",
    }),
  });

  if (!res.ok) {
    const text = await res.text();
    throw new Error(`GHL create contact failed (${res.status}): ${text}`);
  }

  const data = await res.json();
  return data.contact.id;
}

export async function sendSMS(
  contactId: string,
  message: string
): Promise<void> {
  const res = await fetch(`${GHL_BASE}/conversations/messages`, {
    method: "POST",
    headers: getHeaders(),
    body: JSON.stringify({
      type: "SMS",
      contactId,
      message,
    }),
  });

  if (!res.ok) {
    const text = await res.text();
    throw new Error(`GHL send SMS failed (${res.status}): ${text}`);
  }
}
```

```ts
export async function scoreLeads(leads: Lead[]): Promise<ScoreResult[]> {
  const leadsData = leads.map((l) => ({
    phone: l.phone,
    firstName: l.firstName,
    lastName: l.lastName,
    address: `${l.propertyAddress}, ${l.city}, ${l.state} ${l.zip}`,

    sqft: l.sqft,

    absenteeOwner: l.absenteeOwner,
    freeAndClear: l.freeAndClear,
  }));

  const response = await anthropic.messages.create({
    model: "claude-sonnet-4-6",
    max_tokens: 4096,
    messages: [
      {
        role: "user",
        content: `You are a lead scoring assistant for a home remodeling company. Score each lead 1-10 for how likely they are to need and be able to afford home remodeling.

Scoring criteria:
- HIGH (7-10): Older home (15+ years), high equity, owner-occupied, long ownership, larger home
- MEDIUM (4-6): 5-15 year old home, moderate equity, average characteristics
- LOW (1-3): Recently sold, corporate owned, very low equity, vacant

Return ONLY a JSON array. No markdown, no explanation. Each item must have exactly these fields:
- "phone": the lead's phone number (string, exactly as provided)
- "score": integer 1-10
- "reason": one sentence explaining the score

Leads:
${JSON.stringify(leadsData, null, 2)}`,
      },
    ],
  });

  const text =
    response.content[0].type === "text" ? response.content[0].text : "";
  const jsonMatch = text.match(/\[[\s\S]*\]/);

  return JSON.parse(jsonMatch[0]) as ScoreResult[];
}
```

```ts
let defaultDb: Database.Database | null = null;

export function getDb(dbPath?: string): Database.Database {
  if (dbPath === ":memory:") {
    const db = new Database(":memory:");
    initDb(db);
    return db;
  }

  if (!defaultDb) {
    const resolvedPath =
      dbPath || path.join(process.cwd(), "data", "outreach.db");
    defaultDb = new Database(resolvedPath);
    initDb(defaultDb);
  }

  return defaultDb;
}

function initDb(db: Database.Database): void {
  db.exec(`
    CREATE TABLE IF NOT EXISTS send_log (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      leadId TEXT NOT NULL,
      firstName TEXT NOT NULL,
      lastName TEXT NOT NULL,
      phone TEXT NOT NULL,
      address TEXT NOT NULL,
      score INTEGER NOT NULL,
      message TEXT NOT NULL,
      ghlContactId TEXT,
      status TEXT NOT NULL,
      error TEXT,
      sentAt TEXT NOT NULL
    )
  `);
}

export function insertSendLog(
  db: Database.Database,
  entry: SendResult
): void {
  const stmt = db.prepare(`
    INSERT INTO send_log (leadId, firstName, lastName, phone, address, score, message, ghlContactId, status, error, sentAt)
    VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
  `);
  stmt.run(
    entry.leadId,
    entry.firstName,
    entry.lastName,
    entry.phone,
    entry.address,
    entry.score,
    entry.message,
    entry.ghlContactId,
    entry.status,
    entry.error,
    entry.sentAt
  );
}

export function querySendLog(
  db: Database.Database,
  filters?: { status?: string; dateFrom?: string; dateTo?: string }
): SendLogEntry[] {
  let sql = "SELECT * FROM send_log WHERE 1=1";
  const params: string[] = [];

  if (filters?.status) {
    sql += " AND status = ?";
    params.push(filters.status);
  }
  if (filters?.dateFrom) {
    sql += " AND sentAt >= ?";
    params.push(filters.dateFrom);
  }
  if (filters?.dateTo) {
    sql += " AND sentAt <= ?";
    params.push(filters.dateTo);
  }

  sql += " ORDER BY sentAt DESC";

  const stmt = db.prepare(sql);
  return stmt.all(...params) as SendLogEntry[];
}
```

```tsx
interface UploadZoneProps {
  onFileLoaded: (csvText: string) => void;

export default function UploadZone({ onFileLoaded }: UploadZoneProps) {
  const [dragging, setDragging] = useState(false);
  const [fileName, setFileName] = useState<string | null>(null);

  const handleFile = useCallback(
    (file: File) => {
      if (!file.name.endsWith(".csv")) {
        alert("Please upload a CSV file");
        return;
      }
      setFileName(file.name);
      const reader = new FileReader();
      reader.onload = (e) => {
        const text = e.target?.result as string;
        onFileLoaded(text);
      };
      reader.readAsText(file);
    },
    [onFileLoaded]
  );

  const handleDrop = useCallback(
    (e: React.DragEvent) => {
      e.preventDefault();
      setDragging(false);
      const file = e.dataTransfer.files[0];
      if (file) handleFile(file);
    },
    [handleFile]
  );

  const handleChange = useCallback(
    (e: React.ChangeEvent<HTMLInputElement>) => {
      const file = e.target.files?.[0];
      if (file) handleFile(file);
    },
    [handleFile]
  );

  return (
    <div
      onDragOver={(e) => {
        e.preventDefault();
        setDragging(true);
      }}
      onDragLeave={() => setDragging(false)}
      onDrop={handleDrop}
      className={`border-2 border-dashed rounded-xl p-12 text-center cursor-pointer transition-colors ${
        dragging
```

```tsx
  async function scoreAndGenerate(rawLeads: Lead[]) {

        const err = await scoreRes.json();
        throw new Error(err.error || "Scoring failed");
      }

      const scoredLeads: ScoredLead[] = rawLeads.map((lead, i) => {
        const scoreData = scores.find(
          (s: { phone: string }) => s.phone === lead.phone

          ...lead,
          id: `lead-${i}-${lead.phone}`,

          scoreReason: scoreData.reason,
          message: "",
          selected: scoreData.score >= 5,
          status: "new" as const,

              prev.map((l) => (l.id === id ? { ...l, message } : l))
            )
          }
```

```tsx
"use client";

import { useState, useEffect } from "react";
import { useRouter } from "next/navigation";
import SendProgress from "@/components/send-progress";
import type { ScoredLead, SendResult } from "@/lib/types";

export default function SendPage() {
  const router = useRouter();
  const [total, setTotal] = useState(0);
  const [current, setCurrent] = useState(0);
  const [results, setResults] = useState<SendResult[]>([]);
  const [done, setDone] = useState(false);

  useEffect(() => {
    const stored = sessionStorage.getItem("outreach_send");
    if (!stored) {
      router.push("/");
      return;
    }

    const leads: ScoredLead[] = JSON.parse(stored);
    setTotal(leads.length);
    sendLeads(leads);
  }, []);

  async function sendLeads(leads: ScoredLead[]) {
    for (let i = 0; i < leads.length; i++) {
      setCurrent(i + 1);

      try {
        const res = await fetch("/api/send", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ leads: [leads[i]] }),
        });

        const { results: sendResults } = await res.json();
        setResults((prev) => [...prev, ...sendResults]);
      } catch {
        setResults((prev) => [
          ...prev,
          {
            leadId: leads[i].id,
            firstName: leads[i].firstName,
            lastName: leads[i].lastName,
            phone: leads[i].phone,
            address: `${leads[i].propertyAddress}, ${leads[i].city}`,
            score: leads[i].score,
            message: leads[i].message,
            ghlContactId: null,
            status: "failed" as const,
            error: "Network error",
            sentAt: new Date().toISOString(),
          },
        ]);
      }
    }

    sessionStorage.removeItem("outreach_send");
    setDone(true);
  }

  return (
    <div className="py-12">
      <SendProgress
        total={total}
        current={current}
        results={results}
        done={done}
      />

      {done && (
        <div className="text-center mt-8">
          <button
            onClick={() => router.push("/log")}
            className="px-6 py-2 bg-blue-600 text-white font-medium rounded-lg hover:bg-blue-700 transition-colors"
          >
            View Send Log
          </button>
        </div>
      )}
    </div>
  );
}
```

```tsx
"use client";

import { useState, useEffect, useCallback } from "react";
import LogTable from "@/components/log-table";
import type { SendLogEntry } from "@/lib/types";

export default function LogPage() {
  const [entries, setEntries] = useState<SendLogEntry[]>([]);
  const [statusFilter, setStatusFilter] = useState<string>("all");
  const [loading, setLoading] = useState(true);

  const fetchLog = useCallback(async () => {
    setLoading(true);
    const params = new URLSearchParams();
    if (statusFilter !== "all") params.set("status", statusFilter);

    const res = await fetch(`/api/log?${params}`);
    const data = await res.json();
    setEntries(data.entries);
    setLoading(false);
  }, [statusFilter]);

  useEffect(() => {
    fetchLog();
  }, [fetchLog]);

  function exportCSV() {
    const headers = [
      "Date",
      "First Name",
      "Last Name",
      "Phone",
      "Address",
      "Score",
      "Message",
      "Status",
      "Error",
    ];
    const rows = entries.map((e) => [
      e.sentAt,
      e.firstName,
      e.lastName,
      e.phone,
      e.address,
      e.score,
      `"${e.message.replace(/"/g, '""')}"`,
      e.status,
      e.error || "",
    ]);

    const csv = [headers.join(","), ...rows.map((r) => r.join(","))].join("\n");
    const blob = new Blob([csv], { type: "text/csv" });
    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    a.href = url;
    a.download = `outreach-log-${new Date().toISOString().split("T")[0]}.csv`;
    a.click();
    URL.revokeObjectURL(url);
  }

  return (
    <div>
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-bold text-gray-900">Send Log</h1>
        <button
          onClick={exportCSV}
          disabled={entries.length === 0}
          className="px-4 py-2 bg-gray-100 text-gray-700 text-sm font-medium rounded-lg hover:bg-gray-200 transition-colors disabled:opacity-50"
        >
          Export CSV
        </button>
      </div>

      <div className="flex gap-2 mb-4">
        {["all", "sent", "failed"].map((s) => (
          <button
            key={s}
            onClick={() => setStatusFilter(s)}
            className={`px-3 py-1 text-sm rounded-full ${
              statusFilter === s
                ? "bg-blue-600 text-white"
                : "bg-gray-100 text-gray-600 hover:bg-gray-200"
            }`}
          >
            {s.charAt(0).toUpperCase() + s.slice(1)}
          </button>
        ))}
      </div>

      <div className="bg-white rounded-lg border border-gray-200">
        {loading ? (
          <p className="text-center text-gray-500 py-12">Loading...</p>
        ) : (
          <LogTable entries={entries} />
        )}
      </div>
    </div>
  );
}
```
